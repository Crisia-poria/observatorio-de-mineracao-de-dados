    <script>
        // Use a wrapper to ensure lucide is loaded
        document.addEventListener("DOMContentLoaded", () => {
            if (typeof lucide !== 'undefined') {
                lucide.createIcons();
            } else {
                console.error("Lucide not loaded yet.");
                setTimeout(() => {
                    if (typeof lucide !== 'undefined') lucide.createIcons();
                }, 500);
            }

            initNetworkGraph();
            // Initialize Audio Listeners
            initAudioUI();
        });

        // --- Audio System (Vocal Synthesis) ---
        let audioCtx;
        let isAudioInit = false;
        let isAudioUnlocked = false; 

        // Formant Frequencies for Vowels (Approximate)
        // A: 800, 1200 | E: 500, 2300 | I: 300, 2500 | O: 500, 1000 | U: 300, 900
        const formants = {
            'A': { f1: 800, f2: 1200 },
            'E': { f1: 500, f2: 2300 },
            'I': { f1: 300, f2: 2500 },
            'O': { f1: 500, f2: 1000 },
            'U': { f1: 300, f2: 900 }
        };

        // Map Colors/Types to Vowels
        // Tech (Green) -> I (High energy)
        // Finance (Orange) -> O (Hollow)
        // Broker/Infra (Blue) -> U (Deep/Dark)
        // App (Pink) -> A (Open/Loud)
        // State (Purple) -> E (Nasal/Bureaucratic)
        
        // Helper to get vowel based on color hex (since nodes use direct hex)
        function getVowelFromColor(color) {
            if (color === '#00ff41') return 'I'; // Green
            if (color === '#f97316') return 'O'; // Orange
            if (color === '#3b82f6') return 'U'; // Blue
            if (color === '#ec4899') return 'A'; // Pink
            if (color === '#a855f7') return 'E'; // Purple
            return 'A'; // Default
        }

        function initAudio() {
            if (isAudioInit) return;
            const AudioContext = window.AudioContext || window.webkitAudioContext;
            audioCtx = new AudioContext();
            isAudioInit = true;
        }

        function unlockAudio() {
            if (!isAudioInit) initAudio();
            if (audioCtx.state === 'suspended') {
                audioCtx.resume().then(() => { isAudioUnlocked = true; });
            } else {
                isAudioUnlocked = true;
            }
        }

        function playVocalSound(color) {
            if (!isAudioUnlocked || !audioCtx) return;
            if (audioCtx.state === 'suspended') { audioCtx.resume(); return; }

            const vowelKey = getVowelFromColor(color);
            const vowel = formants[vowelKey];

            // Source Oscillator (Glottal source imitation - Sawtooth for buzz)
            const osc = audioCtx.createOscillator();
            osc.type = 'sawtooth';
            // Base pitch varying slightly for organic feel (A2 - C3 range)
            osc.frequency.value = 110 + Math.random() * 20; 

            // Formant Filters
            const filter1 = audioCtx.createBiquadFilter();
            filter1.type = 'bandpass';
            filter1.frequency.value = vowel.f1;
            filter1.Q.value = 5; // Resonance

            const filter2 = audioCtx.createBiquadFilter();
            filter2.type = 'bandpass';
            filter2.frequency.value = vowel.f2;
            filter2.Q.value = 5;

            // Gain Envelope
            const gainNode = audioCtx.createGain();
            
            // Distortion (optional grit)
            const distortion = audioCtx.createWaveShaper();
            function makeDistortionCurve(amount) {
                const k = amount;
                const n_samples = 44100;
                const curve = new Float32Array(n_samples);
                const deg = Math.PI / 180;
                for (let i = 0; i < n_samples; ++i) {
                    const x = i * 2 / n_samples - 1;
                    curve[i] = (3 + k) * x * 20 * deg / (Math.PI + k * Math.abs(x));
                }
                return curve;
            }
            distortion.curve = makeDistortionCurve(50); // Mild distortion
            distortion.oversample = '4x';

            // Routing: Osc -> Distortion -> Parallel Filters -> Gain -> Out
            osc.connect(distortion);
            
            // Connect to both filters in parallel
            distortion.connect(filter1);
            distortion.connect(filter2);

            filter1.connect(gainNode);
            filter2.connect(gainNode);

            gainNode.connect(audioCtx.destination);

            // Envelope
            const now = audioCtx.currentTime;
            gainNode.gain.setValueAtTime(0, now);
            gainNode.gain.linearRampToValueAtTime(0.4, now + 0.05); // Attack
            gainNode.gain.exponentialRampToValueAtTime(0.001, now + 0.4); // Decay

            osc.start(now);
            osc.stop(now + 0.4);
        }

        // Generic click sound for UI buttons (simpler glitch)
        function playClickSound() {
            if (!isAudioUnlocked || !audioCtx) return;
            const osc = audioCtx.createOscillator();
            const gain = audioCtx.createGain();
            osc.type = 'square';
            osc.frequency.value = 80;
            gain.gain.setValueAtTime(0.1, audioCtx.currentTime);
            gain.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + 0.1);
            osc.connect(gain);
            gain.connect(audioCtx.destination);
            osc.start();
            osc.stop(audioCtx.currentTime + 0.1);
        }

        }

        function copyLink() {
            const url = window.location.href;
            navigator.clipboard.writeText(url).then(() => {
                showToast();
            }).catch(err => {
                console.error('Failed to copy: ', err);
                // Fallback for some browsers
                const textArea = document.createElement("textarea");
                textArea.value = url;
                document.body.appendChild(textArea);
                textArea.select();
                document.execCommand("copy");
                document.body.removeChild(textArea);
                showToast();
            });
        }

            function updatePhysics() {
                // Configuração do "Octaedro Curvo" (Losango Côncavo / Superelipse Ajustada)
                // Ajustando para ser mais "fechada" nas pontas (topo/baixo) e mais larga (lados)
                
                const centerX = w / 2;
                const centerY = h / 2;
                
                // Raios independentes para esticar verticalmente
                const radiusX = Math.min(w, h) * 0.35; 
                const radiusY = Math.min(w, h) * 0.45; // Mais alto que largo

                const totalNodes = simulationNodes.length;

                // Rotação orbital lenta
                const time = Date.now() * 0.0001; 
                const rotationOffset = time;

                simulationNodes.forEach((n, i) => {
                    // Distribuição angular
                    let angle = (i / totalNodes) * Math.PI * 2 - Math.PI/2 + rotationOffset;

                    // Superelipse com expoente < 1 para pontas côncavas (fechadas/agudas)
                    // Exponente 0.8 faz uma curva "pinçada" (côncava) lembrando uma estrela ninja ou cristal.
                    const exponent = 0.8; 
                    
                    const absCos = Math.abs(Math.cos(angle));
                    const absSin = Math.abs(Math.sin(angle));
                    const sgnCos = Math.sign(Math.cos(angle));
                    const sgnSin = Math.sign(Math.sin(angle));

                    // Aplica a deformação
                    const targetX = centerX + radiusX * sgnCos * Math.pow(absCos, 2/exponent);
                    const targetY = centerY + radiusY * sgnSin * Math.pow(absSin, 2/exponent);

                    // Física de atração para o ponto alvo
                    const forceStrength = 0.08; 
                    n.vx += (targetX - n.x) * forceStrength;
                    n.vy += (targetY - n.y) * forceStrength;

                    // Repulsão local
                    for(let j=0; j<simulationNodes.length; j++) {
                        if (i === j) continue;
                        let b = simulationNodes[j];
                        let dx = n.x - b.x;
                        let dy = n.y - b.y;
                        let dist = Math.sqrt(dx*dx + dy*dy) || 1;
                        let minDist = n.r + b.r + 30; 

                        if (dist < minDist) {
                            let force = (minDist - dist) / dist;
                            let repulsion = 0.15; 
                            n.vx += (dx/dist) * force * repulsion;
                            n.vy += (dy/dist) * force * repulsion;
                        }
                    }

                    n.vx *= 0.85; // Atrito
                    n.vy *= 0.85;

                    n.x += n.vx;
                    n.y += n.vy;
                });
            }

            function draw() {
                ctx.clearRect(0, 0, w, h);
                let time = Date.now() * 0.0015;

                // Render positions with float
                simulationNodes.forEach(n => {
                    let floatOffset = Math.sin(time + n.phase) * 4; 
                    n.renderX = n.x;
                    n.renderY = n.y + floatOffset;
                });

                // Desenhar Links
                ctx.lineWidth = 1;
                links.forEach(link => {
                    let source = simulationNodes.find(n => n.id === link.source);
                    let target = simulationNodes.find(n => n.id === link.target);
                    
                    if(source && target) {
                        let isConnected = hoveredNode && (hoveredNode.id === source.id || hoveredNode.id === target.id);
                        ctx.beginPath();
                        ctx.moveTo(source.renderX, source.renderY);
                        ctx.lineTo(target.renderX, target.renderY);
                        
                        if (hoveredNode) {
                            ctx.strokeStyle = isConnected ? '#00ff41' : '#222';
                            ctx.globalAlpha = isConnected ? 0.8 : 0.05;
                            ctx.lineWidth = isConnected ? 1.5 : 1;
                        } else {
                            ctx.strokeStyle = '#222';
                            ctx.globalAlpha = 0.4;
                            ctx.lineWidth = 1;
                        }
                        ctx.stroke();
                        ctx.globalAlpha = 1;
                    }
                });

                // Desenhar Crateras
                simulationNodes.forEach(n => {
                    let rx = n.renderX;
                    let ry = n.renderY;
                    let isHovered = hoveredNode && hoveredNode.id === n.id;
                    let isRelated = hoveredNode && links.some(l => 
                        (l.source === hoveredNode.id && l.target === n.id) || 
                        (l.target === hoveredNode.id && l.source === n.id)
                    );
                    let nodeColor = n.color;
                    let opacity = 1;

                    if (hoveredNode && !isHovered && !isRelated) {
                        opacity = 0.2;
                    }

                    // Glow
                    if (isHovered) {
                        ctx.shadowBlur = 30;
                        ctx.shadowColor = nodeColor;
                    } else {
                        ctx.shadowBlur = 0;
                    }

                    // Crater Gradient
                    let grad = ctx.createRadialGradient(rx, ry, n.r * 0.1, rx, ry, n.r);
                    grad.addColorStop(0, 'rgba(0,0,0,1)'); 
                    grad.addColorStop(0.7, 'rgba(0,0,0,0.8)');
                    grad.addColorStop(1, hexToRgba(nodeColor, 0.4)); 

                    ctx.beginPath();
                    ctx.arc(rx, ry, n.r, 0, Math.PI * 2);
                    ctx.fillStyle = grad;
                    ctx.fill();

                    // Rim
                    ctx.beginPath();
                    ctx.arc(rx, ry, n.r, 0, Math.PI * 2);
                    ctx.strokeStyle = nodeColor;
                    ctx.lineWidth = isHovered ? 3 : 2; 
                    ctx.globalAlpha = opacity;
                    ctx.stroke();

                    // Rings
                    ctx.beginPath();
                    ctx.arc(rx, ry, n.r * 0.65, 0, Math.PI * 2);
                    ctx.strokeStyle = nodeColor;
                    ctx.lineWidth = 1;
                    ctx.globalAlpha = opacity * 0.5;
                    ctx.stroke();

                    // Callout Line
                    const dx = rx - w/2;
                    const dy = ry - h/2;
                    const dist = Math.sqrt(dx*dx + dy*dy);
                    const dirX = dx / dist;
                    const dirY = dy / dist;

                    const labelStartDist = n.r + 5;
                    const labelEndDist = n.r + 35; 

                    ctx.globalAlpha = opacity;
                    ctx.shadowBlur = 0;
                    
                    ctx.beginPath();
                    ctx.moveTo(rx + dirX * labelStartDist, ry + dirY * labelStartDist);
                    ctx.lineTo(rx + dirX * labelEndDist, ry + dirY * labelEndDist);
                    ctx.strokeStyle = nodeColor;
                    ctx.lineWidth = 1;
                    ctx.stroke();
                    
                    ctx.beginPath();
                    ctx.arc(rx + dirX * labelEndDist, ry + dirY * labelEndDist, 2, 0, Math.PI*2);
                    ctx.fillStyle = nodeColor;
                    ctx.fill();

                    // TEXTO
                    ctx.fillStyle = '#fff';
                    ctx.font = 'bold 12px "Space Grotesk"';
                    
                    let textAlign = 'center';
                    let textX = rx + dirX * labelEndDist;
                    let textY = ry + dirY * labelEndDist;

                    if (Math.abs(dirX) > 0.2) { 
                        textAlign = dirX > 0 ? 'left' : 'right';
                        textX += dirX > 0 ? 8 : -8;
                    }
                    
                    ctx.textAlign = textAlign;
                    ctx.textBaseline = 'middle';
                    
                    ctx.fillText(n.label, textX, textY - 6);

                    ctx.fillStyle = nodeColor;
                    ctx.font = '10px "JetBrains Mono"';
                    ctx.fillText(n.volume, textX, textY + 6);
                });

                requestAnimationFrame(() => {
                    updatePhysics();
                    draw();
                });
            }

            resize();
            draw();
        }
